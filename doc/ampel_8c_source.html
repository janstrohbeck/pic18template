34 void ampel (void)
35 {
36  // State Variable
37  static uint8_t state = 0;
38  // Delay Counter
39  static uint8_t delays = 0;
40  // Here we later set the number of delays required to get to the next state
41  uint8_t req_delays = 0;
42 
43  // First make sure all LEDs are enabled
44  LED (LEDGREEN, ENABLED);
45  LED (LEDYELLOW, ENABLED);
46  LED (LEDRED, ENABLED);
47  // Then turn them all off
48  LED (LEDGREEN, OFF);
49  LED (LEDYELLOW, OFF);
50  LED (LEDRED, OFF);
51 
52  // Activate Pull-Up-Resistors for PORTB
53  INTCON2 &= ~(1<<7);
54 
55  // Is the Button pressed?
56  if ( state == 0 && !(INPPORT & (1<<INP)) )
57  {
58  // Then start the Traffic Light sequence
59  state = 1;
60  }
61 
62  // Switch over the state
63  // Enable the corresponding LEDs and set the required delays after which the next state shall be triggered
64  switch (state)
65  {
66  case 1:
67  LED (LEDGREEN, ON);
68  req_delays = DELAYS_GREEN;
69  break;
70  case 2:
71  LED (LEDYELLOW, ON);
72  req_delays = DELAYS_YELLOW;
73  break;
74  case 3:
75  LED (LEDRED, ON);
76  req_delays = DELAYS_RED;
77  break;
78  case 4:
79  LED (LEDYELLOW, ON);
80  LED (LEDRED, ON);
81  req_delays = DELAYS_YELLOWRED;
82  break;
83  // Idle state
84  case 0:
85  default:
86  // State 0 is default-state
87  LED (LEDGREEN, ON);
88  break;
89  }
90 
91  // Only check for delays when not in idle state 0
92  if (state != 0)
93  {
94  // If the required delays are reached
95  if (delays == req_delays)
96  {
97  // Go to the next state
98  state++;
99  // And reset the delay counter
100  delays = 0;
101  }
102 
103  // Increment counter
104  delays++;
105 
106  // At the end of the sequence
107  if (state > 4)
108  // Go to idle state again
109  state = 0;
110  }
111 }
